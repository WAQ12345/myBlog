✨ 3D 粒子交互系统 ​

一个基于 Three.js 和 MediaPipe 的 3D 粒子算法与视觉交互项目，实现了高性能的粒子系统和丰富的手势交互功能。

📋 项目简介 ​

这是一个纯前端实现的 3D 粒子交互系统，使用 Three.js 创建 3D 场景和粒子系统，MediaPipe 实现手势识别，支持多种交互方式：

  • 手势控制：🖐️ 张开手掌散开、✊ 握拳合拢、🤏 捏合聚焦照片、👍 切换颜色主题
  • 按钮控制：无摄像头时使用按钮切换模式
  • 照片上传：支持上传照片并显示在 3D 场景中
  • 颜色主题：三种颜色主题（默认金色、淡蓝色、粉红色）可切换
  • 触摸交互：支持触摸旋转和缩放
  • 响应式设计：完美适配 PC 和移动端

🎯 核心功能 ​

1. 3D 粒子系统 ​

  • 使用 Three.js 创建高性能 3D 粒子系统
  • 2000+ 粒子实时渲染（金色、绿色、红色等）
  • 粒子聚合与散开算法
  • 后处理效果（Bloom 发光）

2. 手势识别 ​

  • 基于 MediaPipe 的手势识别
  • 支持四种手势模式：
    • 张开手掌 → 粒子散开
    • 握拳 → 粒子聚合成树形
    • OK 手势 → 聚焦照片
    • 大拇指伸出 → 切换颜色主题（默认/淡蓝色/粉红色循环切换）

3. 颜色主题 ​

  • 默认主题：金色、绿色、红色经典配色
  • 淡蓝色主题：蓝色系配色
  • 粉红色主题：粉色系配色
  • 支持手势或按钮切换，实时更新所有粒子颜色和背景

4. 交互控制 ​

  • 手势控制（有摄像头时）
  • 按钮控制（无摄像头时）
  • 触摸旋转和缩放（移动端）
  • 照片上传和展示
  • 颜色主题切换（手势或按钮）

// 资源加载超时包装函数（支持多个备用CDN依次尝试）
async function loadWithTimeout(
  importPromise,
  timeout = 30000,
  resourceName = "资源",
  fallbackImports = []
) {
  const isMobile =
    /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(
      navigator.userAgent
    );
  // 移动端使用5秒超时
  const actualTimeout = isMobile ? 5000 : timeout;

  // 所有CDN源（主CDN + 备用CDN）
  const allSources = [importPromise, ...fallbackImports];
  const cdnNames = ["jsdelivr", "unpkg", "cdnjs", "bootcdn"];

  updateLoaderText(`正在加载 ${resourceName}...`);

  for (let attempt = 0; attempt < allSources.length; attempt++) {
    const currentImport =
      typeof allSources[attempt] === "function"
        ? allSources[attempt]()
        : allSources[attempt];
    const cdnName =
      attempt === 0 ? "主CDN" : cdnNames[attempt] || `备用CDN${attempt}`;

    const timeoutPromise = new Promise((_, reject) =>
      setTimeout(
        () =>
          reject(
            new Error(
              `${resourceName}加载超时(${actualTimeout / 1000}秒)`
            )
          ),
        actualTimeout
      )
    );

    try {
      updateLoaderText(`正在从${cdnName}加载 ${resourceName}...`);
      const result = await Promise.race([currentImport, timeoutPromise]);
      updateLoaderText("加载中...");
      if (attempt > 0) {
        console.log(`${resourceName}从${cdnName}加载成功`);
      }
      return result;
    } catch (err) {
      console.warn(
        `${resourceName}从${cdnName}加载失败:`,
        err.message || err
      );
      if (attempt < allSources.length - 1) {
        updateLoaderText(
          `${resourceName}加载失败，尝试${
            cdnNames[attempt + 1] || "下一个"
          }CDN...`
        );
        // 等待1秒后尝试下一个CDN
        await new Promise((resolve) => setTimeout(resolve, 1000));
        continue;
      }
      // 所有CDN都失败了
      console.error(`${resourceName}所有CDN加载失败`);
      throw new Error(`${resourceName}所有CDN加载失败，请检查网络连接`);
    }
  }
}

class Particle {
  constructor(mesh, type, isDust = false) {
    this.mesh = mesh;
    this.type = type;
    this.isDust = isDust;

    this.posTree = new THREE.Vector3();
    this.posScatter = new THREE.Vector3();
    this.baseScale = mesh.scale.x;

    const speedMult = type === "PHOTO" ? 0.3 : 2.0;
    this.spinSpeed = new THREE.Vector3(
      (Math.random() - 0.5) * speedMult,
      (Math.random() - 0.5) * speedMult,
      (Math.random() - 0.5) * speedMult
    );

    this.calculatePositions();
  }

  calculatePositions() {
    const h = CONFIG.particles.treeHeight;
    const halfH = h / 2;
    let t = Math.random();
    t = Math.pow(t, 0.8);
    const y = t * h - halfH;
    let rMax = CONFIG.particles.treeRadius * (1.0 - t);
    if (rMax < 0.5) rMax = 0.5;
    const angle = t * 50 * Math.PI + Math.random() * Math.PI;
    const r = rMax * (0.8 + Math.random() * 0.4);
    this.posTree.set(Math.cos(angle) * r, y, Math.sin(angle) * r);

    let rScatter = this.isDust
      ? 12 + Math.random() * 20
      : 8 + Math.random() * 12;
    const theta = Math.random() * Math.PI * 2;
    const phi = Math.acos(2 * Math.random() - 1);
    this.posScatter.set(
      rScatter * Math.sin(phi) * Math.cos(theta),
      rScatter * Math.sin(phi) * Math.sin(theta),
      rScatter * Math.cos(phi)
    );
  }

  update(dt, mode, focusTargetMesh) {
    let target = this.posTree;

    if (mode === "SCATTER") target = this.posScatter;
    else if (mode === "FOCUS") {
      if (this.mesh === focusTargetMesh) {
        // 调整聚焦位置，稍微远离相机，避免过亮
        const desiredWorldPos = new THREE.Vector3(0, 2, 40);
        const invMatrix = new THREE.Matrix4()
          .copy(mainGroup.matrixWorld)
          .invert();
        target = desiredWorldPos.applyMatrix4(invMatrix);
      } else {
        target = this.posScatter;
      }
    }

    const lerpSpeed =
      mode === "FOCUS" && this.mesh === focusTargetMesh ? 5.0 : 2.0;
    this.mesh.position.lerp(target, lerpSpeed * dt);

    if (mode === "SCATTER") {
      this.mesh.rotation.x += this.spinSpeed.x * dt;
      this.mesh.rotation.y += this.spinSpeed.y * dt;
      this.mesh.rotation.z += this.spinSpeed.z * dt;
    } else if (mode === "TREE") {
      this.mesh.rotation.x = THREE.MathUtils.lerp(
        this.mesh.rotation.x,
        0,
        dt
      );
      this.mesh.rotation.z = THREE.MathUtils.lerp(
        this.mesh.rotation.z,
        0,
        dt
      );
      this.mesh.rotation.y += 0.5 * dt;
    }

    if (mode === "FOCUS" && this.mesh === focusTargetMesh) {
      this.mesh.lookAt(camera.position);
    }

    let s = this.baseScale;
    if (this.isDust) {
      s =
        this.baseScale *
        (0.8 + 0.4 * Math.sin(clock.elapsedTime * 4 + this.mesh.id));
      if (mode === "TREE") s = 0;
    } else if (mode === "SCATTER" && this.type === "PHOTO") {
      s = this.baseScale * 2.5;
    } else if (mode === "FOCUS") {
      if (this.mesh === focusTargetMesh) s = 4.5;
      else s = this.baseScale * 0.8;
    }

    this.mesh.scale.lerp(new THREE.Vector3(s, s, s), 4 * dt);
  }
}

function createParticles() {
  const sphereGeo = new THREE.SphereGeometry(0.5, 32, 32);
  const boxGeo = new THREE.BoxGeometry(0.55, 0.55, 0.55);
  const curve = new THREE.CatmullRomCurve3([
    new THREE.Vector3(0, -0.5, 0),
    new THREE.Vector3(0, 0.3, 0),
    new THREE.Vector3(0.1, 0.5, 0),
    new THREE.Vector3(0.3, 0.4, 0),
  ]);
  const candyGeo = new THREE.TubeGeometry(curve, 16, 0.08, 8, false);

  const theme = CONFIG.colorThemes[STATE.colorThemeIndex];

  const goldMat = new THREE.MeshStandardMaterial({
    color: theme.champagneGold,
    metalness: 1.0,
    roughness: 0.1,
    envMapIntensity: 2.0,
    emissive: theme.goldEmissive,
    emissiveIntensity: 0.3,
  });

  const greenMat = new THREE.MeshStandardMaterial({
    color: theme.deepGreen,
    metalness: 0.2,
    roughness: 0.8,
    emissive: theme.greenEmissive,
    emissiveIntensity: 0.2,
  });

  const redMat = new THREE.MeshPhysicalMaterial({
    color: theme.accentRed,
    metalness: 0.3,
    roughness: 0.2,
    clearcoat: 1.0,
    emissive: theme.redEmissive,
  });

  const candyMat = new THREE.MeshStandardMaterial({
    map: caneTexture,
    roughness: 0.4,
  });

  // 保存材质引用
  materialRefs.goldMat = goldMat;
  materialRefs.greenMat = greenMat;
  materialRefs.redMat = redMat;

  for (let i = 0; i < CONFIG.particles.count; i++) {
    const rand = Math.random();
    let mesh, type;

    if (rand < 0.4) {
      mesh = new THREE.Mesh(boxGeo, greenMat);
      type = "BOX";
    } else if (rand < 0.7) {
      mesh = new THREE.Mesh(boxGeo, goldMat);
      type = "GOLD_BOX";
    } else if (rand < 0.82) {
      mesh = new THREE.Mesh(sphereGeo, goldMat);
      type = "GOLD_SPHERE";
    } else if (rand < 0.95) {
      mesh = new THREE.Mesh(sphereGeo, redMat);
      type = "RED";
    } else {
      mesh = new THREE.Mesh(candyGeo, candyMat);
      type = "CANE";
    }

    const s = 0.25 + Math.random() * 0.25;
    mesh.scale.set(s, s, s);
    mesh.rotation.set(
      Math.random() * 6,
      Math.random() * 6,
      Math.random() * 6
    );

    mainGroup.add(mesh);
    particleSystem.push(new Particle(mesh, type, false));
  }

  // 创建五角星几何体
  function createStarGeometry(outerRadius, innerRadius, depth) {
    const shape = new THREE.Shape();
    const points = 5;
    const angleStep = (Math.PI * 2) / points;

    for (let i = 0; i < points * 2; i++) {
      const angle = (i * angleStep) / 2 + Math.PI / 2;
      const radius = i % 2 === 0 ? outerRadius : innerRadius;
      const x = Math.cos(angle) * radius;
      const y = Math.sin(angle) * radius;

      if (i === 0) {
        shape.moveTo(x, y);
      } else {
        shape.lineTo(x, y);
      }
    }
    shape.closePath();

    const extrudeSettings = {
      depth: depth,
      bevelEnabled: true,
      bevelThickness: 0.05,
      bevelSize: 0.05,
      bevelSegments: 3,
    };

    return new THREE.ExtrudeGeometry(shape, extrudeSettings);
  }

  const starGeo = createStarGeometry(1.2, 0.5, 0.3);
  const starTheme = CONFIG.colorThemes[STATE.colorThemeIndex];
  const starMat = new THREE.MeshStandardMaterial({
    color: starTheme.champagneGold,
    emissive: starTheme.champagneGold,
    emissiveIntensity: 1.0,
    metalness: 1.0,
    roughness: 0,
  });
  materialRefs.starMat = starMat;
  const star = new THREE.Mesh(starGeo, starMat);
  star.position.set(0, CONFIG.particles.treeHeight / 2 + 1.2, 0);
  mainGroup.add(star);

  mainGroup.add(photoMeshGroup);
}

function createDust() {
  const geo = new THREE.TetrahedronGeometry(0.08, 0);
  const mat = new THREE.MeshBasicMaterial({
    color: 0xffeebb,
    transparent: true,
    opacity: 0.8,
  });

  for (let i = 0; i < CONFIG.particles.dustCount; i++) {
    const mesh = new THREE.Mesh(geo, mat);
    mesh.scale.setScalar(0.5 + Math.random());
    mainGroup.add(mesh);
    particleSystem.push(new Particle(mesh, "DUST", true));
  }
}

function addPhotoToScene(texture) {
  const frameGeo = new THREE.BoxGeometry(1.4, 1.4, 0.05);
  const currentTheme = CONFIG.colorThemes[STATE.colorThemeIndex];
  const frameMat = new THREE.MeshStandardMaterial({
    color: currentTheme.champagneGold,
    metalness: 1.0,
    roughness: 0.1,
  });
  const frame = new THREE.Mesh(frameGeo, frameMat);

  const photoGeo = new THREE.PlaneGeometry(1.2, 1.2);
  // 使用 MeshBasicMaterial，不受光照影响，直接显示图片原始颜色
  // 通过调整颜色来降低整体亮度
  const photoMat = new THREE.MeshBasicMaterial({
    map: texture,
    toneMapped: false, // 禁用色调映射，保持原始颜色
    color: 0xaaaaaa, // 稍微降低整体亮度（0xaaaaaa 约为 67% 亮度）
  });
  const photo = new THREE.Mesh(photoGeo, photoMat);
  photo.position.z = 0.04;

  const group = new THREE.Group();
  group.add(frame);
  group.add(photo);

  const s = 0.8;
  group.scale.set(s, s, s);

  photoMeshGroup.add(group);
  particleSystem.push(new Particle(group, "PHOTO", false));
}

function handleImageUpload(e) {
  const files = e.target.files;
  if (!files.length) return;
  Array.from(files).forEach((f) => {
    const reader = new FileReader();
    reader.onload = (ev) => {
      new THREE.TextureLoader().load(ev.target.result, (t) => {
        t.colorSpace = THREE.SRGBColorSpace;
        addPhotoToScene(t);
      });
    };
    reader.readAsDataURL(f);
  });
}

// 检测是否有摄像头
async function checkCameraAvailability() {
  try {
    if (!navigator.mediaDevices?.getUserMedia) {
      console.log("getUserMedia not supported");
      return false;
    }
    // 先尝试获取设备列表（需要权限）
    try {
      const devices = await navigator.mediaDevices.enumerateDevices();
      const hasVideoInput = devices.some(
        (device) => device.kind === "videoinput"
      );
      if (hasVideoInput) {
        return true;
      }
    } catch (e) {
      // 如果没有权限，尝试直接请求摄像头
      console.log("Cannot enumerate devices, trying direct access");
    }
    // 尝试直接请求摄像头（移动端可能需要）
    try {
      const stream = await navigator.mediaDevices.getUserMedia({
        video: true,
      });
      stream.getTracks().forEach((track) => track.stop()); // 立即停止，只是检测
      return true;
    } catch (e) {
      console.log("Camera access failed:", e.name);
      return false;
    }
  } catch (err) {
    console.warn("Camera check failed:", err);
    return false;
  }
}

// --- MEDIAPIPE 手势识别 ---
async function initMediaPipe() {
  try {
    video = document.getElementById("webcam");

    // 先检测是否有摄像头
    console.log("Checking camera availability...");
    STATE.hasCamera = await checkCameraAvailability();
    console.log("Camera available:", STATE.hasCamera);

    if (!STATE.hasCamera) {
      console.log("No camera available, using button controls");
      showButtonControls();
      if (video) {
        video.style.display = "none";
      }
      return;
    }

    // 先请求摄像头权限（在 MediaPipe 之前）
    const constraints = {
      video: {
        facingMode: "user",
        width: { ideal: 320 },
        height: { ideal: 240 },
        aspectRatio: { ideal: 4 / 3 },
      },
    };

    let stream;
    try {
      stream = await navigator.mediaDevices.getUserMedia(constraints);
      video.srcObject = stream;
      STATE.cameraEnabled = true;
      console.log("Camera access granted");
    } catch (permissionErr) {
      console.warn("Camera permission denied:", permissionErr);
      STATE.cameraEnabled = false;
      showButtonControls();
      if (video) {
        video.style.display = "none";
      }
      return;
    }

    // 添加超时处理（缩短超时时间，更快回退）
    const timeoutPromise = new Promise((_, reject) =>
      setTimeout(
        () => reject(new Error("MediaPipe initialization timeout")),
        8000
      )
    );

    console.log("Initializing MediaPipe...");
    const visionPromise = FilesetResolver.forVisionTasks(
      "https://cdn.jsdelivr.net/npm/@mediapipe/tasks-vision@0.10.3/wasm"
    );

    const vision = await Promise.race([visionPromise, timeoutPromise]);

    const handLandmarkerPromise = HandLandmarker.createFromOptions(
      vision,
      {
        baseOptions: {
          modelAssetPath: `https://storage.googleapis.com/mediapipe-models/hand_landmarker/hand_landmarker/float16/1/hand_landmarker.task`,
          delegate: "GPU",
        },
        runningMode: "VIDEO",
        numHands: 1,
      }
    );

    handLandmarker = await Promise.race([
      handLandmarkerPromise,
      timeoutPromise,
    ]);
    console.log("MediaPipe initialized successfully");

    // 确保视频已加载后再启动预测
    if (video.readyState >= 2) {
      console.log("Video already loaded, starting gesture recognition");
      predictWebcam();
    } else {
      video.addEventListener(
        "loadeddata",
        () => {
          console.log("Video loaded, starting gesture recognition");
          predictWebcam();
        },
        { once: true }
      );
    }

    // 隐藏按钮控制，使用手势控制
    hideButtonControls();
    console.log("手势识别已启用，按钮控制已隐藏");
  } catch (err) {
    // 静默处理超时错误，直接切换到按钮控制
    if (err.message && err.message.includes("timeout")) {
      console.log(
        "MediaPipe initialization timeout, using button controls"
      );
    } else {
      console.warn("MediaPipe initialization failed:", err);
    }
    STATE.cameraEnabled = false;
    showButtonControls();
    // 隐藏摄像头预览，如果初始化失败
    if (video) {
      video.style.display = "none";
    }
    // 停止摄像头流
    if (video && video.srcObject) {
      const stream = video.srcObject;
      stream.getTracks().forEach((track) => track.stop());
      video.srcObject = null;
    }
  }
}

// 按钮控制模式切换
function setModeFromButton(mode) {
  STATE.mode = mode;
  STATE.focusTarget = null;

  if (mode === "FOCUS") {
    const photos = particleSystem.filter((p) => p.type === "PHOTO");
    if (photos.length) {
      STATE.focusTarget =
        photos[Math.floor(Math.random() * photos.length)].mesh;
    } else {
      // 如果没有照片，切换到树形
      STATE.mode = "TREE";
    }
  }

  // 更新按钮状态
  document.querySelectorAll(".control-btn").forEach((btn) => {
    btn.classList.remove("active");
    if (btn.dataset.mode === mode) {
      btn.classList.add("active");
    }
  });
}

let lastVideoTime = -1;
async function predictWebcam() {
  // 检查视频是否已加载且摄像头已启用
  if (
    !video ||
    !video.readyState ||
    video.readyState < 2 ||
    !STATE.cameraEnabled
  ) {
    requestAnimationFrame(predictWebcam);
    return;
  }

  if (video.currentTime !== lastVideoTime) {
    lastVideoTime = video.currentTime;
    if (handLandmarker && STATE.cameraEnabled) {
      try {
        const result = handLandmarker.detectForVideo(
          video,
          performance.now()
        );
        processGestures(result);
      } catch (err) {
        console.warn("手势识别错误:", err);
      }
    }
  }
  requestAnimationFrame(predictWebcam);
}

// 检测单根手指是否弯曲（使用多个关节点判断）
function isFingerCurled(lm, tipIdx, pipIdx, mcpIdx) {
  const tip = lm[tipIdx];
  const pip = lm[pipIdx];
  const mcp = lm[mcpIdx];
  const wrist = lm[0];

  // 方法1：指尖到手腕的距离 vs 指根到手腕的距离
  const tipToWrist = Math.hypot(
    tip.x - wrist.x,
    tip.y - wrist.y,
    tip.z - wrist.z
  );
  const mcpToWrist = Math.hypot(
    mcp.x - wrist.x,
    mcp.y - wrist.y,
    mcp.z - wrist.z
  );

  // 方法2：检查Y轴（在正常手势下，弯曲的手指tip.y > pip.y）
  // 这里用相对比较，更鲁棒
  const tipBelowPip = tip.y > pip.y + 0.02; // 指尖在PIP关节下方

  // 综合判断：指尖比指根更接近手腕，或者指尖在PIP下方
  return tipToWrist < mcpToWrist * 0.85 || tipBelowPip;
}

// 检测拇指是否弯曲（拇指结构不同，单独处理）
function isThumbCurled(lm) {
  const thumbTip = lm[4];
  const thumbIp = lm[3];
  const thumbMcp = lm[2];
  const indexMcp = lm[5];

  // 拇指尖到食指根部的距离（弯曲时会靠近）
  const thumbToIndex = Math.hypot(
    thumbTip.x - indexMcp.x,
    thumbTip.y - indexMcp.y
  );

  // 拇指尖到拇指根的距离
  const thumbLength = Math.hypot(
    thumbTip.x - thumbMcp.x,
    thumbTip.y - thumbMcp.y
  );

  return thumbToIndex < 0.1 || thumbLength < 0.08;
}

// 检测大拇指是否伸出（点赞手势）
function isThumbUp(lm) {
  const thumbTip = lm[4];
  const thumbIp = lm[3];
  const thumbMcp = lm[2];
  const indexMcp = lm[5];

  // 拇指尖到食指根部的距离（伸出时应该较远）
  const thumbToIndex = Math.hypot(
    thumbTip.x - indexMcp.x,
    thumbTip.y - indexMcp.y
  );

  // 拇指尖到拇指根的距离
  const thumbLength = Math.hypot(
    thumbTip.x - thumbMcp.x,
    thumbTip.y - thumbMcp.y
  );

  // 拇指向上：拇指尖在拇指根上方，且距离食指根部较远
  const thumbUpward = thumbTip.y < thumbMcp.y; // 拇指尖在拇指根上方
  const thumbExtended = thumbLength > 0.12; // 拇指伸直
  const thumbAwayFromIndex = thumbToIndex > 0.15; // 拇指远离食指

  // 其他手指应该弯曲（握拳状态）
  const indexCurled = isFingerCurled(lm, 8, 6, 5);
  const middleCurled = isFingerCurled(lm, 12, 10, 9);
  const ringCurled = isFingerCurled(lm, 16, 14, 13);
  const pinkyCurled = isFingerCurled(lm, 20, 18, 17);
  const otherFingersCurled =
    [indexCurled, middleCurled, ringCurled, pinkyCurled].filter(Boolean)
      .length >= 3;

  return (
    thumbUpward &&
    thumbExtended &&
    thumbAwayFromIndex &&
    otherFingersCurled
  );
}

function processGestures(result) {
  // 只有在摄像头启用且手势识别可用时才处理手势
  if (!STATE.cameraEnabled || !handLandmarker) {
    return;
  }

  if (result.landmarks && result.landmarks.length > 0) {
    STATE.hand.detected = true;
    const lm = result.landmarks[0];
    STATE.hand.x = (lm[9].x - 0.5) * 2;
    STATE.hand.y = (lm[9].y - 0.5) * 2;

    // 检测每根手指是否弯曲
    const indexCurled = isFingerCurled(lm, 8, 6, 5); // 食指
    const middleCurled = isFingerCurled(lm, 12, 10, 9); // 中指
    const ringCurled = isFingerCurled(lm, 16, 14, 13); // 无名指
    const pinkyCurled = isFingerCurled(lm, 20, 18, 17); // 小指
    const thumbCurled = isThumbCurled(lm);

    // 计算弯曲手指数量
    const curledCount = [
      indexCurled,
      middleCurled,
      ringCurled,
      pinkyCurled,
    ].filter(Boolean).length;

    // 捏合检测：拇指和食指靠近
    const thumbTip = lm[4];
    const indexTip = lm[8];
    const pinchDist = Math.hypot(
      thumbTip.x - indexTip.x,
      thumbTip.y - indexTip.y,
      thumbTip.z - indexTip.z
    );
    const isPinching = pinchDist < 0.08; // 放宽阈值

    // 检测OK手势：拇指+食指靠近形成圆圈，其他三指伸直
    const otherFingersStraight =
      !middleCurled && !ringCurled && !pinkyCurled;
    const isOkGesture = isPinching && otherFingersStraight;

    // 检测大拇指伸出（点赞手势）
    const isThumbUpGesture = isThumbUp(lm);

    // 处理颜色主题切换（大拇指手势）
    if (isThumbUpGesture) {
      const now = Date.now();
      if (
        now - STATE.gesture.lastThumbUpTime >=
        STATE.gesture.thumbUpDebounceDelay
      ) {
        STATE.gesture.lastThumbUpTime = now;
        switchColorTheme();
      }
    }

    // 手势判断逻辑（手势控制优先于按钮控制）
    // 1. OK手势（👌）：拇指食指形成圆圈，其他手指伸直 → 聚焦
    // 2. 握拳（✊）：3根以上手指弯曲 → 树形
    // 3. 张开（🖐️）：2根以下手指弯曲 → 散开
    // 4. 大拇指（👍）：伸出大拇指 → 切换颜色主题

    // 检测目标模式
    let targetMode = STATE.mode; // 默认保持当前模式

    if (isOkGesture) {
      targetMode = "FOCUS";
    } else if (curledCount >= 3) {
      targetMode = "TREE";
    } else if (curledCount <= 1 && !isPinching) {
      targetMode = "SCATTER";
    }

    // 防抖处理：只有目标模式与当前模式不同，且持续一定时间后才切换
    const now = Date.now();

    if (targetMode !== STATE.gesture.lastMode) {
      // 手势变化，重置计时器
      STATE.gesture.lastMode = targetMode;
      STATE.gesture.lastModeTime = now;
    } else if (targetMode !== STATE.mode) {
      // 目标模式已稳定，检查是否超过防抖延迟
      if (
        now - STATE.gesture.lastModeTime >=
        STATE.gesture.debounceDelay
      ) {
        // 执行模式切换
        STATE.mode = targetMode;

        if (targetMode === "FOCUS") {
          // OK手势 → 聚焦照片
          const photos = particleSystem.filter((p) => p.type === "PHOTO");
          if (photos.length) {
            STATE.focusTarget =
              photos[Math.floor(Math.random() * photos.length)].mesh;
          } else {
            // 如果没有照片，切换到树形
            STATE.mode = "TREE";
            STATE.focusTarget = null;
          }
        } else {
          STATE.focusTarget = null;
        }

        // 更新按钮状态
        updateButtonStates();
      }
    }
    // 如果目标模式与当前模式相同，不需要处理
  } else {
    // 未检测到手势时，如果之前是手势控制，保持当前模式
    // 不强制切换，让用户继续看到当前状态
    STATE.hand.detected = false;
    // 重置防抖状态
    STATE.gesture.lastMode = STATE.mode;
    STATE.gesture.lastModeTime = Date.now();
  }
}

// 更新按钮状态（同步手势控制的状态）
function updateButtonStates() {
  document.querySelectorAll(".control-btn").forEach((btn) => {
    btn.classList.remove("active");
    if (btn.dataset.mode === STATE.mode) {
      btn.classList.add("active");
    }
  });
}

// 切换颜色主题
function switchColorTheme() {
  STATE.colorThemeIndex =
    (STATE.colorThemeIndex + 1) % CONFIG.colorThemes.length;
  const currentTheme = CONFIG.colorThemes[STATE.colorThemeIndex];

  // 更新背景色和雾效
  scene.background = new THREE.Color(currentTheme.bg);
  scene.fog = new THREE.FogExp2(currentTheme.bg, 0.01);

  // 更新材质颜色
  if (materialRefs.goldMat) {
    materialRefs.goldMat.color.setHex(currentTheme.champagneGold);
    materialRefs.goldMat.emissive.setHex(currentTheme.goldEmissive);
  }
  if (materialRefs.greenMat) {
    materialRefs.greenMat.color.setHex(currentTheme.deepGreen);
    materialRefs.greenMat.emissive.setHex(currentTheme.greenEmissive);
  }
  if (materialRefs.redMat) {
    materialRefs.redMat.color.setHex(currentTheme.accentRed);
    materialRefs.redMat.emissive.setHex(currentTheme.redEmissive);
  }
  if (materialRefs.starMat) {
    materialRefs.starMat.color.setHex(currentTheme.champagneGold);
    materialRefs.starMat.emissive.setHex(currentTheme.champagneGold);
  }

  console.log(`颜色主题已切换为: ${currentTheme.name}`);
}

function setupEvents() {
  window.addEventListener("resize", () => {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
    composer.setSize(window.innerWidth, window.innerHeight);
  });

  document
    .getElementById("file-input")
    .addEventListener("change", handleImageUpload);

  // 按钮控制事件
  document.querySelectorAll(".control-btn").forEach((btn) => {
    btn.addEventListener("click", (e) => {
      const mode = e.target.dataset.mode;
      if (mode) {
        setModeFromButton(mode);
      }
    });
  });

  // 触摸旋转
  const canvas = renderer.domElement;

  canvas.addEventListener(
    "touchstart",
    (e) => {
      if (e.touches.length === 1) {
        STATE.touch.isDragging = true;
        STATE.touch.lastX = e.touches[0].clientX;
        STATE.touch.lastY = e.touches[0].clientY;
      } else if (e.touches.length === 2) {
        STATE.touch.pinchDistance = getPinchDistance(e.touches);
      }
    },
    { passive: true }
  );

  canvas.addEventListener(
    "touchmove",
    (e) => {
      if (e.touches.length === 1 && STATE.touch.isDragging) {
        const deltaX = e.touches[0].clientX - STATE.touch.lastX;
        const deltaY = e.touches[0].clientY - STATE.touch.lastY;

        STATE.rotation.y += deltaX * 0.008;
        STATE.rotation.x += deltaY * 0.004;
        STATE.rotation.x = Math.max(
          -0.5,
          Math.min(0.5, STATE.rotation.x)
        );

        STATE.touch.lastX = e.touches[0].clientX;
        STATE.touch.lastY = e.touches[0].clientY;
      } else if (e.touches.length === 2) {
        const newDistance = getPinchDistance(e.touches);
        const scale = newDistance / STATE.touch.pinchDistance;
        camera.position.z = Math.max(
          25,
          Math.min(80, camera.position.z / scale)
        );
        STATE.touch.pinchDistance = newDistance;
      }
    },
    { passive: true }
  );

  canvas.addEventListener(
    "touchend",
    () => {
      STATE.touch.isDragging = false;
    },
    { passive: true }
  );
}

function getPinchDistance(touches) {
  const dx = touches[0].clientX - touches[1].clientX;
  const dy = touches[0].clientY - touches[1].clientY;
  return Math.sqrt(dx * dx + dy * dy);
}

function animate() {
  requestAnimationFrame(animate);
  const dt = clock.getDelta();

  const autoRotate = !STATE.touch.isDragging;

  // 手势控制旋转
  if (STATE.mode === "SCATTER" && STATE.hand.detected) {
    const targetRotY = STATE.hand.x * Math.PI * 0.9;
    const targetRotX = STATE.hand.y * Math.PI * 0.25;
    STATE.rotation.y += (targetRotY - STATE.rotation.y) * 3.0 * dt;
    STATE.rotation.x += (targetRotX - STATE.rotation.x) * 3.0 * dt;
  } else if (autoRotate) {
    if (STATE.mode === "TREE") {
      STATE.rotation.y += 0.3 * dt;
      STATE.rotation.x += (0 - STATE.rotation.x) * 2.0 * dt;
    } else {
      STATE.rotation.y += 0.1 * dt;
    }
  }

  mainGroup.rotation.y = STATE.rotation.y;
  mainGroup.rotation.x = STATE.rotation.x;

  // 聚焦模式下降低曝光，让图片更清晰可见
  if (STATE.mode === "FOCUS") {
    renderer.toneMappingExposure = THREE.MathUtils.lerp(
      renderer.toneMappingExposure,
      0.65,
      dt * 2
    );
  } else {
    renderer.toneMappingExposure = THREE.MathUtils.lerp(
      renderer.toneMappingExposure,
      2.2,
      dt * 2
    );
  }

  particleSystem.forEach((p) =>
    p.update(dt, STATE.mode, STATE.focusTarget)
  );
  composer.render();
}

🛠️ 技术栈 ​

  • Three.js 0.160.0 - 3D 图形渲染
  • MediaPipe Tasks Vision - 手势识别
  • ES6 Modules - 模块化开发
  • 原生 JavaScript - 无框架依赖

🎮 使用说明 ​

PC 端 ​

  1. 打开页面后，允许摄像头权限
  2. 使用手势控制：
    • 🖐️ 张开手掌 - 粒子散开
    • ✊ 握拳 - 粒子聚合成树形
    • 🤏 OK 手势 - 聚焦照片
    • 👍 伸出大拇指 - 切换颜色主题（保持约 0.5 秒）
  3. 点击"添加照片"上传照片

移动端 ​

  1. 如果摄像头可用，使用手势控制
  2. 如果摄像头不可用，使用底部按钮控制
  3. 支持触摸旋转和缩放

🔧 配置说明 ​

粒子配置 ​

  • count: 2000 - 装饰粒子数量
  • dustCount: 800 - 灰尘粒子数量
  • treeHeight: 20 - 树的高度
  • treeRadius: 6.5 - 树的半径

CDN 配置 ​

项目支持多个备用 CDN，自动切换：

  • jsdelivr（主 CDN）
  • unpkg（备用 1）
  • cdnjs（备用 2）
  • bootcdn（备用 3，国内）

💡 注意事项 ​

  1. 摄像头权限：手势识别需要摄像头权限，请确保浏览器已授权
  2. HTTPS 要求：摄像头功能需要 HTTPS 环境（GitHub Pages 等自动提供）
  3. 网络要求：首次加载需要下载 Three.js 和 MediaPipe 资源，建议使用 WiFi
  4. 浏览器兼容：需要支持 ES6 模块的现代浏览器
